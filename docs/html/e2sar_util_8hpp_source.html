400 static inline const result<std::pair<ip::address, u_int16_t>> string_tuple_to_ip_and_port(const std::string &t) noexcept
401 {
402 // search for last ":" (ip:port) or "]" (ipv6 by itself) whichever comes last
403 auto const pos = t.find_last_of("]:");
404
405 // IPv4 or IPv6 by itself
406 if ((pos == std::string::npos) || (t[pos] == ']'))
407 {
408 auto r1 = string_to_ip(t);
409 if (r1)
410 return std::pair<ip::address, u_int16_t>(r1.value(), 0);
411 else
412 return E2SARErrorInfo{E2SARErrorc::ParameterError, "Unable to convert "s + t + " to ip address and port"s};
413 }
414
415 // port with either IPv4 or IPv6 address x.x.x.x:num or [x:x:x:x::y]:num
416 auto r1 = string_to_ip(t.substr(0, pos));
417 auto r2 = string_to_port(t.substr(pos + 1));
418 if (r1 && r2)
419 return std::pair<ip::address, int>(r1.value(), r2.value());
420 return E2SARErrorInfo{E2SARErrorc::ParameterError, "Unable to convert "s + t + " to ip address and port"s};
421 }

429 static inline result<std::vector<ip::address>> resolveHost(const std::string &host_name) noexcept
430 {
431
432 std::vector<ip::address> addresses;
433 boost::asio::io_service io_service;
434
435 try
436 {
437 ip::udp::resolver resolver(io_service);
438 ip::udp::resolver::query query(host_name, "443");
439 ip::udp::resolver::iterator iter = resolver.resolve(query);
440 ip::udp::resolver::iterator end; // End marker.
441
442 while (iter != end)
443 {
444 ip::udp::endpoint endpoint = *iter++;
445 addresses.push_back(endpoint.address());
446 }
447 return addresses;
448 }
449 catch (...)
450 {
451 // anything happens - we can't find the host
452 return E2SARErrorInfo{E2SARErrorc::NotFound, "Unable to convert "s + host_name + " to ip address"s};
453 }
454 }

457 struct pair_hash {
458 std::size_t operator()(const std::pair<u_int64_t, u_int16_t>& p) const {
459 u_int64_t hash1 = p.first;
460 u_int64_t tmp = p.second;
461 u_int64_t hash2 = tmp | tmp << 16 | tmp << 32 | tmp << 48;
462 return hash1 ^ hash2; // Combine the two hashes
463 }
