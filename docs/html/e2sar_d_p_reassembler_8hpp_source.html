68 // custom comparator of REHdr buffers/segments - placing outside the event queue
69 // item class just not to confuse things - needed for priority queue
70 // orders segments in the increasing order of offset
71 struct SegmentComparator {
72 bool operator() (const Segment &l, const Segment &r) const {
73 REHdr *lhdr = reinterpret_cast<REHdr*>(l.segment);
74 REHdr *rhdr = reinterpret_cast<REHdr*>(r.segment);
75 return lhdr->bufferOffset > rhdr->bufferOffset;
76 }
77 };

131 // receive event queue definitions
132 static const size_t QSIZE{1000};
133 boost::lockfree::queue<EventQueueItem*> eventQueue{QSIZE};
134 std::atomic<size_t> eventQueueDepth{0};
135
136 // push event on the event queue
137 inline void enqueue(EventQueueItem* item) noexcept
138 {
139 if (eventQueue.push(item))
140 eventQueueDepth++;
141 else
142 recvStats.enqueueLoss++; // event lost, queue was full
143 // queue is lock free so we don't lock
144 recvThreadCond.notify_all();
145 }
146 // pop event off the event queue
147 inline EventQueueItem* dequeue() noexcept
148 {
149 EventQueueItem *item{nullptr};
150 auto a = eventQueue.pop(item);
151 if (a)
152 {
153 eventQueueDepth--;
154 return item;
155 } else
156 return nullptr; // queue was empty
157 }
