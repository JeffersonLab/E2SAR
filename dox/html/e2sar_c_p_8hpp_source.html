137 LBManager(const EjfatURI &cpuri, bool validateServer = true,
138 grpc::SslCredentialsOptions opts = grpc::SslCredentialsOptions()) : _cpuri(cpuri)
139 {
140
141 auto cp_host_r = cpuri.get_cpHost();
142 std::string addr_string;
143 if (!cp_host_r.has_error())
144 {
145 // try hostname
146 auto cp_host_v = cp_host_r.value();
147 addr_string = cp_host_v.first + ":"s + std::to_string(cp_host_v.second);
148 }
149 else
150 {
151 // try address
152 auto cp_addr_r = cpuri.get_cpAddr();
153 if (cp_addr_r.has_error())
154 throw E2SARException("Unable to initialize LBManager due to missing CP address in URI");
155 auto cp_addr_v = cp_addr_r.value();
156 addr_string = cp_addr_v.first.to_string() + ":" + std::to_string(cp_addr_v.second);
157 }
158
159 if (cpuri.get_useTls())
160 {
161 grpc::experimental::TlsChannelCredentialsOptions topts;
162 if (!validateServer)
163 {
164 // disable most of server certificate validation
165 std::shared_ptr<grpc::experimental::NoOpCertificateVerifier> verifier = std::make_shared<grpc::experimental::NoOpCertificateVerifier>();
166 topts.set_verify_server_certs(false);
167 topts.set_check_call_host(false);
168 topts.set_certificate_verifier(verifier);
169 _channel = grpc::CreateChannel(addr_string, grpc::experimental::TlsCredentials(topts));
170 }
171 else
172 {
173 // use provided SSL options
174 _channel = grpc::CreateChannel(addr_string, grpc::SslCredentials(opts));
175 }
176 }
177 else
178 {
179 _channel = grpc::CreateChannel(addr_string, grpc::InsecureChannelCredentials());
180 }
181 _stub = LoadBalancer::NewStub(_channel);
182 }

537 static inline int get_PortRange(int source_count) noexcept
538 {
539 // Based on the proto file enum for the load balancer, seen below,
540 // map the max # of sources a backend will see to the PortRange value.
541 // This is necessay to provide the control plane when registering.
542
543 // Handle edge cases
544 if (source_count < 2)
545 {
546 return 0;
547 }
548 else if (source_count > 16384)
549 {
550 return 14;
551 }
552
553 int maxCount = 2;
554 int iteration = 1;
555
556 while (source_count > maxCount)
557 {
558 iteration++;
559 maxCount <<= 1;
560 }
561
562 return iteration;
563 }
